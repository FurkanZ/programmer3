        OleDbConnection bag = new OleDbConnection("Provider=Microsoft.ACE.OLEDB.12.0;Data Source=a.accdb");
        void cek()
        {
            bag.Open();
            OleDbCommand komut = new OleDbCommand("Select * From Ogrenciler order by rnd(-ID*Time())", bag);
            OleDbDataReader oku = komut.ExecuteReader();
            while (oku.Read())
            {
                listBox1.Items.Add(oku["Ad"] + " " + oku["Soyad"]);
               

            }
            bag.Close();
        }
        private void button1_Click(object sender, EventArgs e)
        {
            
            cek();

        ArrayList sýralar = new ArrayList();
        int sayý = 0;
        int kalanlar = 0;

        private void button2_Click(object sender, EventArgs e)
        {
            foreach (Button item in groupBox1.Controls)
            {
                sayý++;
                for (int i = 0; i < sayý; i++)
                {
                    item.Text = listBox1.Items[i].ToString();
                }
                label1.Text = sayý.ToString();
            }
            kalanlar = listBox1.Items.Count - sayý;
            label2.Text = kalanlar.ToString();
            for (int i = listBox1.Items.Count - 1; i >= sayý; i--)
            {
                listBox2.Items.Add(listBox1.Items[i]);
            }
           Form yenisýnýf = new Form();
            yenisýnýf.Size = new Size(924, 605);
            int butonSayisi = 30;
            int sol = 1; //formun sol tarafýndan atanan deðer
            int alt = 50; // formun üst tarafýndan atanan deðer
            int bol; // bolme iþlemindeki amaç formun boyutuna göre butonlarý sýralý bir þekilde görebilmek için
            bol = Convert.ToInt32(Math.Ceiling(Math.Sqrt(butonSayisi)));
            for (int i = 0; i < butonSayisi; i++)  // girilen buton sayýsýna göre döngü þartý saðlanana kadar oluþturmakta
            {
                Button btn = new Button();
                btn.Name = i.ToString();
                btn.AutoSize = false;
                btn.Size = new Size(yenisýnýf.Width / bol, yenisýnýf.Height / (bol * 2));
                btn.Text = this.listBox2.Items[i].ToString();            
                btn.Font = new Font(btn.Font.FontFamily.Name, 10 );
                btn.Location = new Point(sol, alt);
                btn.FlatStyle = FlatStyle.Flat;
                yenisýnýf.Controls.Add(btn);
                sol += btn.Width + 5;
                if (sol + this.Width / bol > this.Width) // bunu yapmasaydýk butonlar yan yana dizilir alt satýra geçmedi
                {
                    sol = 1;
                    alt += this.Height / (bol * 2) + 5;
                }             
            }
            yenisýnýf.Show();
           
        }
